void Ív(int szín, int méret, int irány)
{
    int párosPáratlan = (méret % 2 == 1? 0 : 1);
    for (int i = 1; i > -3; i -= 2)
    {
        Lépcső(szín, 2, false, i * irány);
        Lépcső(szín, ((méret / 2) - (i == 1? 0 : 1)) - (((méret / 2) - (i == 1 ? 0 : 1)) % 1) + 1 - (méret % 2 == 1 ? 0 : 1), true, i * irány);
        Balra();
        ElőreRakj(párosPáratlan, szín);
        párosPáratlan = 0;
        Balra();
    }

}
void HázikóAlja(int szín, int szélesség, int magasság, int ablakX, int ablakY, int ablakMagasság, int ablakSzélesség, int ajtóX, int ajtóMagasság, int ajtóSzélesség)

    Jobbra();

    Négyszög(szín, ablakMagasság, ablakSzélesség);
    Előre(-ablakY);
    Balra();
    ElőreRakj(ajtóX - 1 - ablakX, szín);
    Előre();
    Jobbra();
    Négyszög(szín, ajtóMagasság, ajtóSzélesség);

    Jobbra();
    ElőreRakj(szélesség - 1, szín);
    Jobbra();
    ElőreRakj(magasság - 1, szín);
    Jobbra(2);
}

void Kerítés(int szín, int db, int szélesség, int magasság, int rúd1, int rúd2)
{
    for (int i = 0; i < db; i++)
    {
        Előre(rúd1 - 1);
        Jobbra();
        ElőreRakj(szélesség, szín);
        Előre();
        Jobbra();
        ElőreRakj(rúd1 - 1, szín);
        Jobbra(2);
        Előre(rúd1 - 2);
        ElőreRakj(magasság - rúd1 + 1, szín);
        Előre(-(magasság - rúd2));
        Balra();
        ElőreRakj(szélesség, szín);
        Előre(-szélesség);
        Jobbra();
        Előre(-(magasság - (magasság - rúd2)) + 1);
    }
    Balra();
    Előre((szélesség + 1) * db);
    Jobbra();
}
void KerítésesHázikó(int alapSzín, int szélesség, int magasság, int ablakX, int ablakY, int ablakMagasság, int ablakSzélesség, int ajtóX, int ajtóMagasság, int ajtóSzélesség, int tetőSzín, int perem, int kerítésSzín, int kerítésDB, int kerítésSzélesség, int kerítésMagasság, int rúd1, int rúd2)
{
    Házikó(alapSzín, szélesség, magasság, ablakX, ablakY, ablakMagasság, ablakSzélesség, ajtóX, ajtóMagasság, ajtóSzélesség, tetőSzín, perem);
    Kerítés(kerítésSzín, kerítésDB, kerítésSzélesség, kerítésMagasság, rúd1, rúd2);
    Balra();
    Előre(szélesség - 1);
    Jobbra();
}
